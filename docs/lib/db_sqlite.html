  43def __filter_str(s, charclass='a-zA-Z0-9_'):
  44    """
  45    Filter a string to keep only allowed characters.
  46
  47    This function removes all characters from a string except those matching the allowed
  48    character class. By default, it allows only alphanumeric characters (`a-z`, `A-Z`, `0-9`)
  49    and underscores (`_`), making the output safe for use in variable names, table names,
  50    index names, and similar identifiers.
  51
  52    ### Parameters
  53    - **s** (`str`):
  54      The input string to sanitize.
  55    - **charclass** (`str`, optional):
  56      A regex character class defining allowed characters.
  57      Defaults to `'a-zA-Z0-9_'`.
  58
  59    ### Returns
  60    - **str**:
  61      A sanitized string containing only characters matching the allowed character class.
  62
  63    ### Notes
  64    - Useful for cleaning user input before using it in database object names or variable names.
  65    - The function uses regular expressions for filtering.
  66
  67    ### Example
  68    >>> __filter_str('user@example.ch')
  69    'userexamplech'
  70
  71    >>> __filter_str('project-123', charclass='a-zA-Z0-9')
  72    'project123'
  73    """
  74    regex = f'[^{charclass}]'
  75    return re.sub(regex, '', s)

  78def __sha1sum(string):
  79    """
  80    Calculate the SHA-1 hash of a given string.
  81
  82    This function encodes the input as bytes (if necessary) and returns its SHA-1 checksum
  83    as a hexadecimal string.
  84
  85    ### Parameters
  86    - **string** (`str`):
  87      The input string to hash.
  88
  89    ### Returns
  90    - **str**:
  91      The SHA-1 hash of the input string, represented as a 40-character hexadecimal string.
  92
  93    ### Notes
  94    - Internally, the input is safely converted to bytes before hashing using `txt.to_bytes()`.
  95    - SHA-1 produces a fixed-size 160-bit (20-byte) hash, commonly used for checksums and
  96      identifiers.
  97
  98    ### Example
  99    >>> __sha1sum('linuxfabrik')
 100    '74301e766db4a4006ec1fbd6e031760e7e322223'
 101    """
 102    return hashlib.sha1(txt.to_bytes(string)).hexdigest()

 105def close(conn):
 106    """
 107    Close a SQLite database connection safely.
 108
 109    This function attempts to close an open database connection.
 110    It does not automatically commit any uncommitted changes â€” if you close the connection
 111    without calling `commit()` first, any uncommitted changes will be lost.
 112
 113    ### Parameters
 114    - **conn** (`sqlite3.Connection` or compatible):
 115      An active database connection object.
 116
 117    ### Returns
 118    - **bool**:
 119      - `True` if the connection was closed successfully.
 120      - `False` if an exception occurred during closing.
 121
 122    ### Notes
 123    - Always call `commit()` manually before calling `close()` if you want to save changes.
 124    - Exceptions during closing are caught and handled silently.

 127    >>> close(conn)
 128    True
 129    """
 130    try:
 131        conn.close()
 132        return True
 133    except Exception:
 134        return False

 173def compute_load(conn, sensorcol, datacols, count, table='perfdata'):
 174    """
 175    Calculate per-second load metrics based on historical data in a SQLite table.
 176
 177    This function calculates `Load1` (over the last 1 interval) and `Loadn` (over the last `count` intervals)
 178    for one or more sensors, based on timestamped performance data.
 179
 180    ### Parameters
 181    - **conn** (`sqlite3.Connection`):
 182      An active database connection object.
 183    - **sensorcol** (`str`):
 184      Column name that identifies the sensor (e.g., `'interface'`).
 185    - **datacols** (`list` of `str`):
 186      List of columns for which to calculate per-second loads (e.g., `['tx_bytes', 'rx_bytes']`).
 187    - **count** (`int`):
 188      Number of historical entries to use for calculating `Loadn`.
 189    - **table** (`str`, optional):
 190      Name of the table containing the performance data.
 191      Defaults to `'perfdata'`.
 192
 193    ### Returns
 194    - **tuple** (`bool`, `list or bool or str`):
 195      - First element (`bool`): `True` if the calculation succeeded, `False` if a database error occurred.
 196      - Second element:
 197        - A `list` of dictionaries containing per-sensor load values on success.
 198        - `False` if there is not enough data to compute the load.
 199        - Error message (`str`) on database failure.
 200
 201    ### Notes
 202    - The table must contain a `timestamp` column (UNIX epoch seconds).
 203    - Data must exist for each sensor with at least `count` historical entries.
 204    - Results include:
 205      - `<column>1`: Load computed between the two most recent entries.
 206      - `<column>n`: Load computed between the most recent and the oldest of `count` entries.
 207    - Load values are calculated as delta per second.
 208    - Table names are sanitized to allow only safe characters.
 209
 210    ### Example
 211    Calculate loads for `tx_bytes` and `rx_bytes` over 5 intervals:
 212    >>> compute_load(conn, sensorcol='interface', datacols=['tx_bytes', 'rx_bytes'], count=5, table='perfdata')
 213
 214    Example output:
 215
 216        [
 217            {
 218                'interface': 'mgmt1',
 219                'tx_bytes1': 6906,
 220                'rx_bytes1': 10418,
 221                'tx_bytesn': 7442,
 222                'rx_bytesn': 10871
 223            },
 224            ...
 225        ]
 226    """
 227    table = __filter_str(table)
 228
 229    sql = f'SELECT DISTINCT {sensorcol} FROM {table} ORDER BY {sensorcol} ASC;'
 230    success, sensors = select(conn, sql)
 231    if not success:
 232        return False, sensors
 233    if len(sensors) == 0:
 234        return True, False
 235
 236    load = []
 237
 238    for sensor in sensors:
 239        sensor_name = sensor[sensorcol]
 240        success, perfdata = select(
 241            conn,
 242            f'SELECT * FROM {table} WHERE {sensorcol} = :{sensorcol} ORDER BY timestamp DESC;',
 243            data={sensorcol: sensor_name}
 244        )
 245        if not success:
 246            return False, perfdata
 247        if len(perfdata) < count:
 248            return True, False
 249
 250        load1_delta = perfdata[0]['timestamp'] - perfdata[1]['timestamp']
 251        loadn_delta = perfdata[0]['timestamp'] - perfdata[count-1]['timestamp']
 252
 253        tmp = {sensorcol: sensor_name}
 254        for key in datacols:
 255            if key in perfdata[0]:
 256                tmp[f'{key}1'] = (perfdata[0][key] - perfdata[1][key]) / load1_delta if load1_delta else 0
 257                tmp[f'{key}n'] = (perfdata[0][key] - perfdata[count-1][key]) / loadn_delta if loadn_delta else 0
 258        load.append(tmp)
 259
 260    return True, load

 263def connect(path='', filename=''):
 264    """
 265    Connect to a SQLite database file.
 266
 267    This function establishes a connection to a SQLite database file.
 268    If no path is provided, a temporary directory is used.
 269    If no filename is provided, the default filename `'linuxfabrik-monitoring-plugins-sqlite.db'`
 270    is used.
 271
 272    ### Parameters
 273    - **path** (`str`, optional):
 274      Path to the directory containing the database file.
 275      Defaults to the system temporary directory (e.g., `/tmp`).
 276    - **filename** (`str`, optional):
 277      Name of the database file.
 278      Defaults to `'linuxfabrik-monitoring-plugins-sqlite.db'`.
 279
 280    ### Returns
 281    - **tuple** (`bool`, `Connection or str`):
 282      - First element (`bool`): `True` if connection succeeded, `False` if it failed.
 283      - Second element (`Connection` or `str`):
 284        - Database connection object on success.
 285        - Error message string on failure.
 286
 287    ### Notes
 288    - The connection uses a `Row` factory, allowing rows to behave like dictionaries.
 289    - The connection registers a `REGEXP` SQL function for regular expression support.
 290    - Always check the returned success flag before using the connection.
 291
 292    ### Example
 293    >>> success, conn = connect()
 294    >>> if success:
 295    >>>     # Use conn
 296    >>>     pass
 297    >>> else:
 298    >>>     print(conn)
 299    """
 300    def get_filename(path='', filename=''):
 301        """Helper to build the absolute path to the SQLite database file."""
 302        if not path:
 303            path = disk.get_tmpdir()
 304        if not filename:
 305            filename = 'linuxfabrik-monitoring-plugins-sqlite.db'
 306        return os.path.join(path, filename)
 307
 308    db = get_filename(path, filename)
 309
 310    try:
 311        conn = sqlite3.connect(db)
 312        conn.row_factory = sqlite3.Row
 313        conn.text_factory = str
 314        conn.create_function('REGEXP', 2, regexp)
 315        return True, conn
 316    except Exception as e:
 317        return False, f'Connecting to DB {db} failed, Error: {e}'

 320def create_index(conn, column_list, table='perfdata', unique=False, delete_db_on_operational_error=True):
 321    """
 322    Create an index on one or more columns in a SQLite table.
 323
 324    This function creates a (unique or non-unique) index on the specified columns of a table.
 325    If the database structure has changed and an `OperationalError` occurs, the database file
 326    can optionally be deleted automatically.
 327
 328    ### Parameters
 329    - **conn** (`sqlite3.Connection`):
 330      An active database connection object.
 331    - **column_list** (`str`):
 332      A comma-separated list of columns to index, for example `'col1, col2'`.
 333    - **table** (`str`, optional):
 334      The table name. Defaults to `'perfdata'`.
 335    - **unique** (`bool`, optional):
 336      If `True`, creates a unique index.
 337      If `False`, creates a standard (non-unique) index. Defaults to `False`.
 338    - **delete_db_on_operational_error** (`bool`, optional):
 339      If `True`, deletes the database file when an `OperationalError` occurs.
 340      Defaults to `True`.
 341
 342    ### Returns
 343    - **tuple** (`bool`, `bool or str`):
 344      - First element (`bool`): `True` if the operation succeeded, `False` if it failed.
 345      - Second element (`bool` or `str`):
 346        - `True` on success.
 347        - Error message (`str`) describing the failure.
 348
 349    ### Notes
 350    - The table name is sanitized to only allow safe characters.
 351    - The index name is automatically generated as `idx_<sha1sum>`, based on table and column names.
 352    - Index creation uses `IF NOT EXISTS` to avoid errors if the index already exists.
 353
 354    ### Example
 355    >>> create_index(conn, 'hostname, service')
 356    (True, True)
 357
 358    >>> create_index(conn, 'timestamp', table='logs', unique=True)
 359    (True, True)
 360    """
 361    table = __filter_str(table)
 362    index_name = f"idx_{__sha1sum(table + column_list)}"
 363    if unique:
 364        sql = f'CREATE UNIQUE INDEX IF NOT EXISTS {index_name} ON "{table}" ({column_list});'
 365    else:
 366        sql = f'CREATE INDEX IF NOT EXISTS {index_name} ON "{table}" ({column_list});'
 367
 368    c = conn.cursor()
 369    try:
 370        c.execute(sql)
 371        return True, True
 372    except sqlite3.OperationalError as e:
 373        if delete_db_on_operational_error:
 374            rm_db(conn)
 375        return False, f'Operational Error: {e}, Query: {sql}'
 376    except Exception as e:
 377        return False, f'Query failed: {sql}, Error: {e}'

 380def create_table(conn, definition, table='perfdata', drop_table_first=False):
 381    """
 382    Create a database table if it does not exist.
 383
 384    This function creates a table in the SQLite database based on the given column definition.
 385    Optionally, the table can be dropped first if it already exists.
 386
 387    ### Parameters
 388    - **conn** (`sqlite3.Connection`):
 389      An active database connection object.
 390    - **definition** (`str`):
 391      Column definitions for the table, e.g., `'col1 TEXT, col2 INTEGER NOT NULL'`.
 392    - **table** (`str`, optional):
 393      Name of the table to create. Defaults to `'perfdata'`.
 394    - **drop_table_first** (`bool`, optional):
 395      If `True`, drops the table before creating it. Defaults to `False`.
 396
 397    ### Returns
 398    - **tuple** (`bool`, `bool or str`):
 399      - First element (`bool`): `True` if the table was created successfully, `False` if an
 400        error occurred.
 401      - Second element (`bool` or `str`):
 402        - `True` on success.
 403        - Error message (`str`) describing the failure.
 404
 405    ### Notes
 406    - The table name is sanitized to allow only safe characters.
 407    - If `drop_table_first=True`, the function will attempt to drop the existing table before
 408      creating it.
 409    - The table creation uses `IF NOT EXISTS` to avoid errors if the table already exists.
 410
 411    ### Example
 412    Create a new table with three columns:
 413    >>> create_table(conn, 'a TEXT, b TEXT, c INTEGER NOT NULL', table='test')
 414
 415    Resulting SQL:
 416
 417        CREATE TABLE IF NOT EXISTS "test" (a TEXT, b TEXT, c INTEGER NOT NULL);
 418    """
 419    table = __filter_str(table)
 420
 421    if drop_table_first:
 422        success, result = drop_table(conn, table)
 423        if not success:
 424            return success, result
 425
 426    sql = f'CREATE TABLE IF NOT EXISTS "{table}" ({definition});'
 427
 428    c = conn.cursor()
 429    try:
 430        c.execute(sql)
 431        return True, True
 432    except Exception as e:
 433        return False, f'Query failed: {sql}, Error: {e}'

 436def cut(conn, table='perfdata', _max=5, delete_db_on_operational_error=True):
 437    """
 438    Keep only the latest records in a SQLite table, based on `rowid`.
 439
 440    This function deletes older rows from a table, keeping only the most recent `_max` entries
 441    according to the SQLite built-in `rowid`. Useful for maintaining lightweight, capped tables.
 442
 443    ### Parameters
 444    - **conn** (`sqlite3.Connection`):
 445      An active database connection object.
 446    - **table** (`str`, optional):
 447      Name of the table to prune. Defaults to `'perfdata'`.
 448    - **_max** (`int`, optional):
 449      Number of most recent records to keep. Defaults to `5`.
 450    - **delete_db_on_operational_error** (`bool`, optional):
 451      If `True`, deletes the database file when an `OperationalError` occurs.
 452      Defaults to `True`.
 453
 454    ### Returns
 455    - **tuple** (`bool`, `bool or str`):
 456      - First element (`bool`): `True` if deletion succeeded, `False` if it failed.
 457      - Second element (`bool` or `str`):
 458        - `True` on success.
 459        - Error message (`str`) describing the failure.
 460
 461    ### Notes
 462    - The function relies on the implicit `rowid` column for ordering.
 463    - The table name is sanitized to allow only safe characters.
 464    - If an `OperationalError` occurs (e.g., due to schema mismatch), the database file can 
 465      be deleted automatically.
 466    - Uses `LIMIT -1 OFFSET :_max` to delete everything after the most recent `_max` records.
 467
 468    ### Example
 469    >>> cut(conn, table='logs', _max=1000)
 470    (True, True)
 471    """
 472    table = __filter_str(table)
 473
 474    sql = f'''
 475        DELETE FROM {table}
 476        WHERE rowid IN (
 477            SELECT rowid FROM {table}
 478            ORDER BY rowid DESC
 479            LIMIT -1 OFFSET :_max
 480        );
 481    '''
 482
 483    c = conn.cursor()
 484    try:
 485        c.execute(sql, {'_max': _max})
 486        return True, True
 487    except sqlite3.OperationalError as e:
 488        if delete_db_on_operational_error:
 489            rm_db(conn)
 490        return False, f'Operational Error: {e}, Query: {sql}'
 491    except Exception as e:
 492        return False, f'Query failed: {sql}, Error: {e}'

 495def delete(conn, sql, data=None, delete_db_on_operational_error=True):
 496    """
 497    Execute a DELETE command against a SQLite table.
 498
 499    This function deletes records from a table based on the given SQL DELETE statement.
 500    If no WHERE clause is provided, all records are deleted.
 501    Parameter binding is supported for safety.
 502
 503    ### Parameters
 504    - **conn** (`sqlite3.Connection`):
 505      An active database connection object.
 506    - **sql** (`str`):
 507      The SQL DELETE statement to execute.
 508      Use placeholders (`:key`) for parameterized queries.
 509    - **data** (`dict`, optional):
 510      Dictionary of parameters to bind to the SQL statement.
 511      Defaults to an empty dict (no parameters).
 512    - **delete_db_on_operational_error** (`bool`, optional):
 513      If `True`, deletes the database file when an `OperationalError` occurs.
 514      Defaults to `True`.
 515
 516    ### Returns
 517    - **tuple** (`bool`, `int or str`):
 518      - First element (`bool`): `True` if the delete succeeded, `False` if it failed.
 519      - Second element (`int` or `str`):
 520        - Number of rows affected (`int`) on success.
 521        - Error message (`str`) on failure.
 522
 523    ### Notes
 524    - If the WHERE clause is omitted, all rows in the table will be deleted.
 525    - Always use a WHERE clause carefully to avoid unintended full table deletion.
 526    - On schema-related `OperationalError`, the database file can be deleted automatically.
 527
 528    ### Example
 529    Delete records older than a specific timestamp:
 530    >>> sql = 'DELETE FROM logs WHERE timestamp < :cutoff'
 531    >>> data = {'cutoff': 1700000000}
 532    >>> delete(conn, sql, data)
 533    (True, 42)
 534    """
 535    if data is None:
 536        data = {}
 537
 538    c = conn.cursor()
 539    try:
 540        if data:
 541            rowcount = c.execute(sql, data).rowcount
 542        else:
 543            rowcount = c.execute(sql).rowcount
 544        return True, rowcount
 545    except sqlite3.OperationalError as e:
 546        if delete_db_on_operational_error:
 547            rm_db(conn)
 548        return False, f'Operational Error: {e}, Query: {sql}, Data: {data}'
 549    except Exception as e:
 550        return False, f'Query failed: {sql}, Error: {e}, Data: {data}'

 621def get_tables(conn):
 622    """
 623    List all user-defined tables in the SQLite database.
 624

 626    excluding SQLite internal tables (e.g., those starting with `'sqlite_'`).

 631
 632    ### Returns
