 34def coe(result, state=STATE_UNKNOWN):
 35    """
 36    Continue or Exit (CoE)
 37
 38    This function simplifies error handling for function calls that return a `(success, result)`
 39    tuple. If the operation fails, it sanitizes and prints the error message and exits with a given
 40    state. Otherwise, it returns the successful result and allows the script to continue.
 41
 42    ### Parameters
 43    - **result** (`tuple`): A two-element tuple returned from a function.
 44      - `result[0]` (`bool`): Success indicator (`True` if successful, `False` otherwise).
 45      - `result[1]` (`any`): The actual result or an error message.
 46    - **state** (`int`, optional): Exit code to use if the function fails.
 47      Defaults to `STATE_UNKNOWN` (3).
 48
 49    ### Returns
 50    - **any type**: The second element of the result tuple (`result[1]`) if successful.
 51
 52    ### Notes
 53    - Sensitive information in error messages is automatically redacted before printing.
 54    - This function is intended to be used **only** inside the `main()` function of a plugin,
 55      not inside library functions.
 56    - If the function fails (`result[0]` is `False`), the script immediately exits after printing
 57      the sanitized message.
 58
 59    ### Example
 60    Without `coe`:
 61    >>> success, html = lib.url.fetch(URL)
 62    >>> if not success:
 63    >>>     print(html)
 64    >>>     sys.exit(STATE_UNKNOWN)
 65
 66    With `coe`:
 67    >>> html = lib.base.coe(lib.url.fetch(URL))
 68    """
 69    if result[0]:
 70        # success
 71        return result[1]
 72    # getting something like `(False, 'Error message')`; hide passwords in error message
 73    print(txt.sanitize_sensitive_data(result[1]))
 74    sys.exit(state)

 77def cu(msg=None):
 78    """
 79    See you (cu)
 80
 81    Print an optional error message and stack trace, then exit with STATE_UNKNOWN.
 82
 83    This function prints an optional sanitized message, attaches a stack trace if an error occurred,
 84    and exits the script with `STATE_UNKNOWN`. It ensures output is safe for display in web GUIs
 85    by replacing `<` and `>` characters.
 86
 87    ### Parameters
 88    - **msg** (`str`, optional): An optional message to print before exiting.
 89      If provided, it will be stripped, sanitized, and printed.
 90
 91    ### Returns
 92    - **None**: This function does not return; it always exits the script with `STATE_UNKNOWN`.
 93
 94    ### Notes
 95    - If a traceback exists, it is included for debugging, with `<` and `>` replaced by `'`.
 96    - Sensitive information in the message is automatically redacted before printing.
 97    - If no traceback is present, only the optional message (if any) is printed.
 98
 99    ### Example
100    >>> cu("Unable to connect to server")
101
102    >>> cu()
103    """
104    tb = format_exc()
105    has_traceback = tb and 'NoneType: None' not in tb
106
107    if msg is not None:
108        msg = txt.sanitize_sensitive_data(msg).strip()
109        print(msg, end='')
110        print(' (Traceback for debugging purposes attached)\n' if has_tb else '\n')
111
112    if has_traceback:
113        print(tb.replace('<', "'").replace('>', "'"))
114
115    sys.exit(STATE_UNKNOWN)

118def get_perfdata(label, value, uom=None, warn=None, crit=None, _min=None, _max=None):
119    """
120    Returns a Nagios performance data string in the format:  
121    `'label'=value[UOM];[warn];[crit];[min];[max]`
122
123    ### Parameters
124    - **label** (`str`): The name of the performance data label.
125    - **value** (`int` or `float`): The measured value.
126    - **uom** (`str`, optional): The unit of measurement (e.g., 's', 'B', '%'). Defaults to None.
127    - **warn** (`int` or `float`, optional): Warning threshold. Defaults to None.
128    - **crit** (`int` or `float`, optional): Critical threshold. Defaults to None.
129    - **_min** (`int` or `float`, optional): Minimum value. Defaults to None.
130    - **_max** (`int` or `float`, optional): Maximum value. Defaults to None.
131
132    ### Returns
133    - **str**: A properly formatted Nagios performance data string.
134
135    ### Example
136    >>> get_perfdata('load1', 0.42, '', 1.0, 5.0, 0, 10)
137    "'load1'=0.42;1.0;5.0;0;10 "
138    """
139    msg = f"'{label}'={value}{uom or ''};"
140    msg += f'{warn};' if warn is not None else ';'
141    msg += f'{crit};' if crit is not None else ';'
142    msg += f'{_min};' if _min is not None else ';'
143    msg += f'{_max}' if _max is not None else ''
144    return msg + ' '

147def get_state(value, warn, crit, _operator='ge'):
148    """
149    Returns the STATE by comparing `value` to the given thresholds using
150    a comparison `_operator`. `warn` and `crit` thresholds may also be `None`.
151
152    ### Parameters
153    - **value** (`float`): Numeric value to evaluate.
154    - **warn** (`float`): Numeric warning threshold.
155    - **crit** (`float`): Numeric critical threshold.
156    - **_operator** (`str`): Comparison operator to use:  
157      - `eq`: equal to  
158      - `ge`: greater or equal  
159      - `gt`: greater than  
160      - `le`: less or equal  
161      - `lt`: less than  
162      - `ne`: not equal to  
163      - `range`: match Nagios range definition
164
165    ### Returns
166    - **int**: `STATE_OK`, `STATE_WARN`, or `STATE_CRIT`.
167
168    ### Example
169    >>> get_state(15, 10, 20, 'ge')
170    1  # STATE_WARN
171
172    >>> get_state(10, 10, 20, 'gt')
173    0  # STATE_OK
174    """
175    # make sure to use float comparison
176    value = float(value)
177    if _operator == 'ge':
178        if crit is not None:
179            if value >= float(crit):
180                return STATE_CRIT
181        if warn is not None:
182            if value >= float(warn):
183                return STATE_WARN
184        return STATE_OK
185
186    if _operator == 'gt':
187        if crit is not None:
188            if value > float(crit):
189                return STATE_CRIT
190        if warn is not None:
191            if value > float(warn):
192                return STATE_WARN
193        return STATE_OK
194
195    if _operator == 'le':
196        if crit is not None:
197            if value <= float(crit):
198                return STATE_CRIT
199        if warn is not None:
200            if value <= float(warn):
201                return STATE_WARN
202        return STATE_OK
203
204    if _operator == 'lt':
205        if crit is not None:
206            if value < float(crit):
207                return STATE_CRIT
208        if warn is not None:
209            if value < float(warn):
210                return STATE_WARN
211        return STATE_OK
212
213    if _operator == 'eq':
214        if crit is not None:
215            if value == float(crit):
216                return STATE_CRIT
217        if warn is not None:
218            if value == float(warn):
219                return STATE_WARN
220        return STATE_OK
221
222    if _operator == 'ne':
223        if crit is not None:
224            if value != float(crit):
225                return STATE_CRIT
226        if warn is not None:
227            if value != float(warn):
228                return STATE_WARN
229        return STATE_OK
230
231    if _operator == 'range':
232        if crit is not None:
233            if not coe(match_range(value, crit)):
234                return STATE_CRIT
235        if warn is not None:
236            if not coe(match_range(value, warn)):
237                return STATE_WARN
238        return STATE_OK
239
240    return STATE_UNKNOWN

243def get_table(data, cols, header=None, strip=True, sort_by_key=None, sort_order_reverse=False):
244    """
245    Takes a list of dictionaries, formats the data, and returns
246    the formatted data as a text table.
247
248    Inspired by:  
249    https://www.calazan.com/python-function-for-displaying-a-list-of-dictionaries-in-table-format/
250
251    ### Parameters
252    - **data** (`list`): Data to process (list of dictionaries).
253    - **cols** (`list`): List of keys/columns to include from the dictionaries.
254    - **header** (`list`, optional): Custom table headers. Defaults to None.
255    - **strip** (`bool`, optional): If True, strip/trim values. Defaults to True.
256    - **sort_by_key** (`str`, optional): The key to sort the table by. Defaults to None.
257    - **sort_order_reverse** (`bool`, optional): If True, sort descending. Defaults to False.
258
259    ### Returns
260    - **str**: A formatted text table as a single string.
261
262    ### Example
263    >>> data = [{'name': 'Alice', 'age': 30}, {'name': 'Bob', 'age': 25}]
264    >>> cols = ['name', 'age']
265    >>> print(get_table(data, cols))
266    name  ! age
267    ------+----
268    Alice ! 30
269    Bob   ! 25
270    """
271    if not data:
272        return ''
273
274    if sort_by_key:
275        data = sorted(data, key=operator.itemgetter(sort_by_key), reverse=sort_order_reverse)
276
277    if header:
278        data.insert(0, dict(zip(cols, header)))
279
280    column_widths = collections.OrderedDict(
281        (col, max(len(str(row.get(col, '')).strip() if strip else str(row.get(col, '')))
282                  for row in data))
283        for col in cols
284    )
285
286    if header:
287        data.insert(1, {col: '-' * width for col, width in column_widths.items()})
288
289    lines = []
290    for idx, row in enumerate(data):
291        sep = ' ! ' if idx != 1 else '-+-'
292        lines.append(sep.join(f'{row[col]:<{column_widths[col]}}' for col in cols))
293
294    return '\n'.join(lines) + '\n'

297def get_worst(state1, state2):
298    """
299    Compares `state1` to `state2` and returns the worse state based on the following priority:  
300    STATE_OK < STATE_UNKNOWN < STATE_WARNING < STATE_CRITICAL  
301    It will prioritize any non-OK state.
302
303    Note that numerically the priority order does not match their integer values.
304
305    ### Parameters
306    - **state1** (`int`): The first state to compare.
307    - **state2** (`int`): The second state to compare.
308
309    ### Returns
310    - **int**: The worse state according to the priority order.
311
312    ### Example
313    >>> get_worst(STATE_OK, STATE_WARNING)
314    STATE_WARNING
315
316    >>> get_worst(STATE_UNKNOWN, STATE_CRITICAL)
317    STATE_CRITICAL
318    """
319    state1 = int(state1)
320    state2 = int(state2)
321    if STATE_CRIT in (state1, state2):
322        return STATE_CRIT
323    if STATE_WARN in (state1, state2):
324        return STATE_WARN
325    if STATE_UNKNOWN in (state1, state2):
326        return STATE_UNKNOWN
327    return STATE_OK

330def guess_type(v, consumer='python'):
331    """
332    Guess the type of a value (None, int, float, or string) for different types of consumers
333    (e.g., Python, SQLite).
334
335    For Python, it returns the actual value converted to its type (`int`, `float`, or `str`).
336    For SQLite, it returns a string describing the type (`'integer'`, `'real'`, `'text'`).
337
338    ### Parameters
339    - **v** (`any`): The value to guess the type for.
340    - **consumer** (`str`, optional): The consumer type ('python' or 'sqlite'). Defaults to
341      'python'.
342
343    ### Returns
344    - **any**: 
345      - If `consumer='python'`, returns `None`, `int`, `float`, or `str`.
346      - If `consumer='sqlite'`, returns `'integer'`, `'real'`, or `'text'`.
347
348    ### Example
349    >>> guess_type('1')
350    1
351
352    >>> guess_type('1', 'sqlite')
353    'integer'
354
355    >>> guess_type('1.0')
356    1.0
357
358    >>> guess_type('1.0', 'sqlite')
359    'real'
360
361    >>> guess_type('abc')
362    'abc'
363
364    >>> guess_type('abc', 'sqlite')
365    'text'
366
367    >>> value_type = lib.base.guess_type(value)
368    >>> if isinstance(value_type, int) or isinstance(value_type, float):
369    >>>     ...
370    """
371    if v is None:
372        return None if consumer == 'python' else 'text'
373
374    try:
375        int(v)
376        return int(v) if consumer == 'python' else 'integer'
377    except (ValueError, TypeError):
378        try:
379            float(v)
380            return float(v) if consumer == 'python' else 'real'
381        except (ValueError, TypeError):
382            return str(v) if consumer == 'python' else 'text'

425def lookup_lod(haystack, key, needle):
426    """
427    Search in a list of dictionaries ("lod") for a key containing a specific value
428    and return the first dictionary item found.
429
430    Returns `(index, item)` if the needle was found, otherwise `(-1, None)`.
431
432    ### Parameters
433    - **haystack** (`list`): A list of dictionaries to search through.
434    - **key** (`str`): The key to look for in each dictionary.
435    - **needle** (`any`): The value to match against the specified key.
436
437    ### Returns
438    - **tuple**: 
439        - If found: (index, dictionary item).
440        - If not found: (-1, None).
441
442    ### Example
443    >>> haystack = [
444    ...     {"name": "Tom", "age": 10},
445    ...     {"name": "Mark", "age": 5},
446    ...     {"name": "Pam", "age": 7},
447    ...     {"name": "Dick", "age": 12}
448    ... ]
449    >>> lookup_lod(haystack, 'name', 'Pam')
450    (2, {'name': 'Pam', 'age': 7})
451
452    >>> lookup_lod(haystack, 'name', 'Pamela')
453    (-1, None)
454    """
455    for idx, item in enumerate(haystack):
456        if isinstance(item, dict) and key in item and item[key] == needle:
457            return idx, item
458    return -1, None

461def match_range(value, spec):
462    """
463    Decides if `value` is inside or outside the Nagios threshold specification.
464
465    ### Parameters
466    - **value** (`int` or `float`): The numeric value to check.
467    - **spec** (`str`): The Nagios range specification string.
468
469    ### Returns
470    - **bool**: 
471      - True if `value` is inside the bounds for a non-inverted `spec`, or outside the bounds for an inverted `spec`.
472      - Otherwise, False.
473
474    ### Example
475    >>> match_range(15, '10')
476    0 10 False
477
478    >>> match_range(15, '-10')
479    (False, 'Start 0 must not be greater than end -10')
480
481    >>> match_range(15, '10:')
482    10 inf False
483
484    >>> match_range(15, ':')
485    0 inf False
486
487    >>> match_range(15, '~:10')
488    -inf 10 False
489
490    >>> match_range(15, '10:20')
491    10 20 False
492
493    >>> match_range(15, '@10')
494    0 10 True
495
496    >>> match_range(15, '@~:20')
497    -inf 20 True
498
499    >>> match_range(15, '@')
500    0 inf True
501    """
502    def parse_range(spec):
503        """
504        Inspired by https://github.com/mpounsett/nagiosplugin/blob/master/nagiosplugin/range.py
505
506        +--------+-------------------+-------------------+--------------------------------+
507        | -w, -c | OK if result is   | WARN/CRIT if      | lib.base.parse_range() returns |
508        +--------+-------------------+-------------------+--------------------------------+
509        | 10     | in (0..10)        | not in (0..10)    | (0, 10, False)                 |
510        +--------+-------------------+-------------------+--------------------------------+
511        | -10    | in (-10..0)       | not in (-10..0)   | (0, -10, False)                |
512        +--------+-------------------+-------------------+--------------------------------+
513        | 10:    | in (10..inf)      | not in (10..inf)  | (10, inf, False)               |
514        +--------+-------------------+-------------------+--------------------------------+
515        | :      | in (0..inf)       | not in (0..inf)   | (0, inf, False)                |
516        +--------+-------------------+-------------------+--------------------------------+
517        | ~:10   | in (-inf..10)     | not in (-inf..10) | (-inf, 10, False)              |
518        +--------+-------------------+-------------------+--------------------------------+
519        | 10:20  | in (10..20)       | not in (10..20)   | (10, 20, False)                |
520        +--------+-------------------+-------------------+--------------------------------+
521        | @10:20 | not in (10..20)   | in 10..20         | (10, 20, True)                 |
522        +--------+-------------------+-------------------+--------------------------------+
523        | @~:20  | not in (-inf..20) | in (-inf..20)     | (-inf, 20, True)               |
524        +--------+-------------------+-------------------+--------------------------------+
525        | @      | not in (0..inf)   | in (0..inf)       | (0, inf, True)                 |
526        +--------+-------------------+-------------------+--------------------------------+
527        """
528        def parse_atom(atom, default):
529            if atom == '':
530                return default
531            return float(atom) if '.' in atom else int(atom)
532
533        if spec is None or str(spec).lower() == 'none':
534            return True, None
535        if not isinstance(spec, str):
536            spec = str(spec)
537
538        invert = spec.startswith('@')
539        if invert:
540            spec = spec[1:]
541
542        start, end = ('', spec)
543        if ':' in spec:
544            parts = spec.split(':', 1)
545            if len(parts) != 2:
546                return False, 'Not using range definition correctly'
547            start, end = parts
548
549        start = float('-inf') if start == '~' else parse_atom(start, 0)
550        end = parse_atom(end, float('inf'))
551
552        if start > end:
553            return False, f'Start {start} must not be greater than end {end}'
554
555        return True, (start, end, invert)
556
557    # workaround for https://github.com/Linuxfabrik/monitoring-plugins/issues/789
558    if isinstance(spec, str):
559        spec = spec.lstrip('\\')
560
561    if spec is None or str(spec).lower() == 'none':
562        return True, True
563
564    success, result = parse_range(spec)
565    if not success:
566        return success, result
567
568    start, end, invert = result
569
570    if isinstance(value, (str, bytes)):
571        value = float(str(value).replace('%', ''))
572
573    if value < start or value > end:
574        return True, not invert
575    return True, invert

578def oao(msg, state=STATE_OK, perfdata='', always_ok=False):
579    """
580    Over and Out (OaO)
581
582    Print a sanitized plugin message with optional performance data and exit the script.
583
584    This function formats and prints a plugin message, appends performance data if provided,
585    sanitizes sensitive information, replaces reserved `|` characters, and exits with the
586    specified state code. Optionally, it can always exit with `STATE_OK` regardless of the given
587    state.
588
589    ### Parameters
590    - **msg** (`str`): The plugin message to print. Will be stripped, sanitized, and processed.
591    - **state** (`int`, optional): The exit code to use. Defaults to `STATE_OK`.
592    - **perfdata** (`str`, optional): Performance data to append after a `|` separator.  
593      Defaults to an empty string (no performance data).
594    - **always_ok** (`bool`, optional): If `True`, forces the exit code to `STATE_OK` regardless
595      of the specified `state`. Defaults to `False`.
596
597    ### Returns
598    - **None**: This function does not return; it terminates the script via `sys.exit()`.
599
600    ### Notes
601    - Any `|` characters inside the message are replaced with `!` to avoid breaking Nagios plugin
602      output format.
603    - Sensitive information like passwords, tokens, and keys is automatically redacted.
604    - `perfdata`, if provided, must follow monitoring plugin standards for performance metrics.
605
606    ### Example
607    >>> oao("Service is healthy", STATE_OK, "load=0.12;1.00;5.00", always_ok=False)
608    Service is healthy|load=0.12;1.00;5.00
609    (and exits with code 0)
610
611    >>> oao("password=secret123 found!", STATE_CRITICAL)
612    password=****** found!
613    (and exits with code 2)
614
615    """
616    msg = txt.sanitize_sensitive_data(msg.strip()).replace('|', '!')
617    if always_ok and msg:
618        parts = msg.split('\n', 1)  # Instead of splitlines(), we just split('\n', 1), so only first line is touched.
619        parts[0] += ' (always ok)'
620        msg = '\n'.join(parts)
621    print(f'{msg}|{perfdata.strip()}' if perfdata else msg)
622    sys.exit(STATE_OK if always_ok else state)

658def sort(array, reverse=True, sort_by_key=False):
659    """
660    Sort a dict, list, or tuple.
661
662    - If dict: sorts by values (default) or keys (if sort_by_key=True).
663    - If list or tuple: sorts the elements.
664    - Other types: returned unchanged.
665
666    When a dictionary is provided, this function returns a list of (key, value)
667    tuples sorted based on the specified criteria:
668      - If `sort_by_key` is False (default), the dictionary items are sorted by their values.
669      - If `sort_by_key` is True, the items are sorted by their keys (compared case-insensitively).
670
671    The sort order is descending by default (`reverse=True`).  
672    If the input is not a dictionary, the original input is returned unmodified.
673
674    ### Parameters
675    - **array** (`dict` or `any`): The dictionary to be sorted. If not a dictionary, the input is returned as is.
676    - **reverse** (`bool`, optional): If True, sort in descending order; if False, ascending. Defaults to True.
677    - **sort_by_key** (`bool`, optional): If True, sort by dictionary keys; if False, by values. Defaults to False.
