144# The "Status" property is common to many Redfish schema, and contains:
145#
146#   Health: This represents the health state of this resource in the absence
147#           of its dependent resources
148#   * Critical  A critical condition exists that requires immediate attention.
149#   * OK        Normal.
150#   * Warning   A condition exists that requires attention
151#
152#   HealthRollup: This represents the overall health state from the view of this
153#                 resource
154#   * Critical  A critical condition exists that requires immediate attention.
155#   * OK        Normal.
156#   * Warning   A condition exists that requires attention.
157#
158#   State:
159#   * Absent                This function or resource is not present or not detected.
160#   * Deferring             The element will not process any commands but will queue new
161#                           requests.
162#   * Disabled              This function or resource has been disabled.
163#   * Enabled               This function or resource has been enabled.
164#   * InTest                This function or resource is undergoing testing.
165#   * Quiesced              The element is enabled but only processes a restricted set of
166#                           commands.
167#   * StandbyOffline        This function or resource is enabled, but awaiting an external action to
168#                           activate it.
169#   * StandbySpare          This function or resource is part of a redundancy set and is awaiting a
170#                           failover or other external action to activate it.
171#   * Starting              This function or resource is starting.
172#   * UnavailableOffline    This function or resource is present but cannot be used.
173#   * Updating              The element is updating and may be unavailable or degraded.

176def get_chassis(redfish):
177    """
178    Extract chassis information from a Redfish API response.
179
180    This function retrieves specific chassis details from a Redfish response and returns
181    them as a dictionary.
182
183    ### Parameters
184    - **redfish** (`dict`): A dictionary containing Redfish chassis data.
185
186    ### Returns
187    - **dict**: A dictionary containing the following chassis details:
188      - **AssetTag** (`str`): The asset tag of the chassis.
189      - **ChassisType** (`str`): The type of the chassis.
190      - **Id** (`str`): The ID of the chassis.
191      - **IndicatorLED** (`str`): The status of the indicator LED.
192      - **Manufacturer** (`str`): The manufacturer of the chassis.
193      - **Model** (`str`): The model of the chassis.
194      - **PartNumber** (`str`): The part number of the chassis.
195      - **PowerState** (`str`): The power state of the chassis (e.g., "On").
196      - **SerialNumber** (`str`): The serial number of the chassis.
197      - **SKU** (`str`): The SKU of the chassis.
198      - **Sensors_@odata.id** (`str`): The sensors' OData ID.
199      - **Status_State** (`str`): The state of the chassis (e.g., "Enabled").
200      - **Status_Health** (`str`): The health status of the chassis (e.g., "OK").
201      - **Status_HealthRollup** (`str`): The health rollup status of the chassis (e.g., "OK").
202
203    ### Example
204    >>> redfish_data = {'AssetTag': '12345', 'ChassisType': 'Rackmount', 'Id': '1', 'PowerState': 'On'}
205    >>> get_chassis(redfish_data)
206    {'AssetTag': '12345', 'ChassisType': 'Rackmount', 'Id': '1', 'PowerState': 'On', ...}
207    """
208    data = {key: redfish.get(key, '') for key in CHASSIS_KEYS}
209    for output_key, (parent_key, child_key) in CHASSIS_NESTED_KEYS.items():
210        data[output_key] = redfish.get(parent_key, {}).get(child_key, '')
211    return data

454def get_manager_logservices_sel_entries(redfish):
455    """
456    Fetch and format SEL (System Event Log) entries from the Redfish API.
457
458    This function retrieves log entries from the Redfish API, processes each entry based on its
459    severity, and formats them into a message string. It also determines the overall state of the
460    log entries (OK, WARNING, CRITICAL).
461
462    ### Parameters
463    - **redfish** (`dict`): A dictionary containing the Redfish log entries under the 'Members' key.
464
465    ### Returns
466    - **tuple**:
467      - **msg** (`str`): A formatted string containing the log entries, including the created time,
468        message, and severity state.
469      - **state** (`int`): The worst state across all log entries, which can be one of the
470        following:
471        - `STATE_OK` (0): All logs are OK.
472        - `STATE_WARN` (1): Some logs are warnings.
473        - `STATE_CRIT` (2): Some logs are critical.
474
475    ### Example
476    >>> redfish_data = {
477    ...     'Members': [
478    ...         {'Created': '2021-08-01', 'Message': 'Temperature is high', 'Severity': 'Critical'},
479    ...         {'Created': '2021-08-02', 'Message': 'Fan speed normal', 'Severity': 'OK'}
480    ...     ]
481    ... }
482    >>> get_manager_logservices_sel_entries(redfish_data)
483    ('* 2021-08-01: Temperature is high [CRITICAL]\n', 2)
484    """
485    lines = []
486    state = STATE_OK
487    for entry in redfish.get('Members', []):
488        severity = entry.get('Severity', '').lower()
489        if severity == 'ok':
490            continue
491        msg_state = SEVERITY_TO_STATE.get(severity, STATE_OK)
492        lines.append(
493            '* {}: {}{}'.format(
494                entry.get('Created', ''),
495                entry.get('Message', ''),
496                base.state2str(msg_state, prefix=' ')
497            )
498        )
499        state = base.get_worst(state, msg_state)
500    return '\n'.join(lines) + ('\n' if lines else ''), state
