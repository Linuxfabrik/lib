 21def bits2human(n, decimals=1, space=False):
 22    """
 23    Converts a given number of bits to a human-readable string, with f-string formatting.
 24
 25    ### Parameters
 26    - **n** (`int` or `float`): The number of bits to convert.
 27    - **decimals** (`int`, optional): Number of decimal places. Defaults to 1.
 28    - **space** (`bool`, optional): If True, adds a space between the value and unit.
 29      Defaults to False.
 30
 31    ### Returns
 32    - **str**: A string like '1.0KiB' or '1.0 KiB'.
 33
 34    ### Example
 35    >>> bits2human(8192)
 36    '1.0KiB'
 37
 38    >>> bits2human(8192, decimals=2)
 39    '1.00KiB'
 40
 41    >>> bits2human(8192, space=True)
 42    '1.0 KiB'
 43
 44    >>> bits2human(8192, decimals=3, space=True)
 45    '1.000 KiB'
 46    """
 47    symbols = ('B', 'KiB', 'MiB', 'GiB', 'TiB', 'PiB', 'EiB', 'ZiB', 'YiB')
 48    prefix = {'B': 8}
 49    for i, symbol in enumerate(symbols[1:], 1):
 50        prefix[symbol] = 1024 ** i * 8
 51
 52    for symbol in reversed(symbols):
 53        if n >= prefix[symbol]:
 54            value = float(n) / prefix[symbol]
 55            sep = ' ' if space else ''
 56            return f'{value:.{decimals}f}{sep}{symbol}'
 57
 58    sep = ' ' if space else ''
 59    return f'{n:.{decimals}f}{sep}{symbols[0]}'

 68    format, using appropriate units like bits per second, kilobits per second, megabits per second,
 69    etc., depending on the size of the input.
 70
 71    ### Parameters
 72    - **n** (`int` or `float`): The number of bits per second to convert.
 73    - **decimals** (`int`, optional): Number of decimal places. Defaults to 1.

 78      unit (e.g., '72Mbps').

 82    '72.0Mbps'
 83
 84    >>> bps2human(72000000, decimals=0)

 91    prefix = {symbol: 1000 ** i for i, symbol in enumerate(symbols)}

103def bytes2human(n, decimals=1, space=False):
104    """
105    Converts a given number of bytes to a human-readable format (e.g., B, KiB, MiB, etc.).
106
107    This function converts an integer number of bytes into a more readable format, using
108    appropriate units such as bytes, kilobytes, megabytes, gigabytes, etc., depending on the size
109    of the input.
110
111    ### Parameters
112    - **n** (`int` or `float`): The number of bytes to convert.
113    - **decimals** (`int`, optional): Number of decimal places. Defaults to 1.
114    - **space** (`bool`, optional): Whether to add a space between value and unit.
115      Defaults to False.
116
117    ### Returns
118    - **str**: The human-readable representation of the input bytes with the appropriate unit
119      (e.g., '1.0KiB').
120
121    ### Example
122    >>> bytes2human(1023)
123    '1023.0B'
124
125    >>> bytes2human(1024)
126    '1.0KiB'
127
128    >>> bytes2human(1536, decimals=2)
129    '1.50KiB'
130
131    >>> bytes2human(1048576, decimals=1, space=True)
132    '1.0 MiB'
133    """
134    symbols = ('B', 'KiB', 'MiB', 'GiB', 'TiB', 'PiB', 'EiB', 'ZiB', 'YiB')
135    prefix = {symbol: 1 << (i + 1) * 10 for i, symbol in enumerate(symbols[1:])}
136
137    for symbol in reversed(symbols[1:]):
138        if n >= prefix[symbol]:
139            value = float(n) / prefix[symbol]
140            sep = ' ' if space else ''
141            return f'{value:.{decimals}f}{sep}{symbol}'
142
143    sep = ' ' if space else ''
144    return f'{n:.{decimals}f}{sep}{symbols[0]}'

147def extract_hrnumbers(s, boundaries=None):
148    """
149    Extracts all numbers from a string that start with a digit and end with a known boundary.
150
151    This function scans the input string and extracts substrings that start with a digit and end
152    with a known boundary character (such as 's', 'm', 'h', etc.), and returns these substrings
153    as a list.
154
155    ### Parameters
156    - **s** (`str`): The input string to extract numbers from.
157    - **boundaries** (`list`, optional): A list of boundary characters that signify the end of
158      a number. Defaults to ['s', 'm', 'h', 'D', 'W', 'M', 'Y'].
159
160    ### Returns
161    - **list**: A list of strings representing the extracted numbers along with their boundaries.
162
163    ### Example
164    >>> string = '31Y 20M7s  88  abc12xyz   4s 5'
165    >>> extract_hrnumbers(string)
166    ['31Y', '20M', '7s', '4s']
167
168    >>> string = '17G 3M 4B'
169    >>> extract_hrnumbers(string, boundaries=['G', 'M', 'B'])
170    ['17G', '3M', '4B']
171    """
172    if boundaries is None:
173        boundaries = ['s', 'm', 'h', 'D', 'W', 'M', 'Y']
174
175    extracted = []
176    start_idx = None
177
178    for idx, char in enumerate(s):
179        if char.isdigit() and start_idx is None:
180            start_idx = idx
181        elif char in boundaries and start_idx is not None:
182            extracted.append(s[start_idx:idx + 1])
183            start_idx = None
184        elif not char.isdigit():
185            start_idx = None
186
187    return extracted

190def human2bytes(string, binary=True):
191    """
192    Converts a human-readable string to the equivalent number of bytes.
193
194    This function converts a string representation of a file size, such as '3.072GiB' or '3.072GB', 
195    into the corresponding number of bytes. It supports both binary (base 1024) and decimal
196    (base 1000) units.
197
198    ### Parameters
199    - **string** (`str`): A string representing the size to convert. It can include any of the
200      common size units like 'GiB', 'GB', 'MB', 'kB', etc.
201    - **binary** (`bool`, optional): If True (default), the function will use binary units
202      (base 1024). If False, it will use decimal units (base 1000).
203
204    ### Returns
205    - **int**: The equivalent size in bytes, or 0 if the conversion fails.
206
207    ### Example
208    >>> human2bytes('3.072GiB')
209    3298534883
210
211    >>> human2bytes('3.072G', binary=False)
212    3072000000
213    """
214    try:
215        string = string.lower()
216        if 'kib' in string:
217            return int(float(string.replace('kib', '').strip()) * 1024)
218        if 'mib' in string:
219            return int(float(string.replace('mib', '').strip()) * 1024**2)
220        if 'gib' in string:
221            return int(float(string.replace('gib', '').strip()) * 1024**3)
222        if 'tib' in string:
223            return int(float(string.replace('tib', '').strip()) * 1024**4)
224        if 'pib' in string:
225            return int(float(string.replace('pib', '').strip()) * 1024**5)
226        if 'eib' in string:
227            return int(float(string.replace('eib', '').strip()) * 1024**6)
228        if 'zib' in string:
229            return int(float(string.replace('zib', '').strip()) * 1024**7)
230        if 'yib' in string:
231            return int(float(string.replace('yib', '').strip()) * 1024**8)
232
233        base = 1024 if binary else 1000
234        if 'k' in string:  # matches "kb" or "k"
235            return int(float(string.replace('kb', '').replace('k', '').strip()) * base)
236        if 'm' in string:
237            return int(float(string.replace('mb', '').replace('m', '').strip()) * base**2)
238        if 'g' in string:
239            return int(float(string.replace('gb', '').replace('g', '').strip()) * base**3)
240        if 't' in string:
241            return int(float(string.replace('tb', '').replace('t', '').strip()) * base**4)
242        if 'p' in string:
243            return int(float(string.replace('pb', '').replace('p', '').strip()) * base**5)
244        if 'e' in string:
245            return int(float(string.replace('eb', '').replace('e', '').strip()) * base**6)
246        if 'z' in string:
247            return int(float(string.replace('zb', '').replace('z', '').strip()) * base**7)
248        if 'y' in string:
249            return int(float(string.replace('yb', '').replace('y', '').strip()) * base**8)
250        if 'b' in string:
251            return int(float(string.replace('b', '').strip()))
252        return 0
253    except Exception:
254        return 0

257def human2seconds(string):
258    """
259    Converts a human-readable duration into seconds.
260
261    This function converts durations given in a human-readable format (e.g., '2.5h', '26Y')
262    into the corresponding number of seconds. The units supported are: years (Y), months (M),
263    weeks (W), days (D), hours (h), minutes (m), and seconds (s).
264
265    ### Parameters
266    - **string** (`str`): A string representing the duration. It should include a numeric
267      value followed by one of the supported units ('Y', 'M', 'W', 'D', 'h', 'm', or 's').
268
269    ### Returns
270    - **int**: The equivalent duration in seconds, rounded down to the nearest integer.
271      Returns 0 if invalid.
272
273    ### Example
274    >>> human2seconds('26Y')
275    819936000
276
277    >>> human2seconds('26M')
278    62899200
279
280    >>> human2seconds('26W')
281    15724800
282
283    >>> human2seconds('26D')
284    2246400
285
286    >>> human2seconds('26h')
287    93600
288
289    >>> human2seconds('26m')
290    1560
291
292    >>> human2seconds('26s')
293    26
294
295    >>> human2seconds('1.5W')
296    907200
297
298    >>> lib.human.human2seconds('0.5D')
299    43200
300
301    >>> human2seconds('2.5h')
302    9000
303
304    >>> human2seconds('a7.3X')
305    0
306
307    >>> human2seconds('invalid')
308    0
309    """
310    unit_to_seconds = {
311        's': 1,
312        'm': 60,
313        'h': 3600,
314        'D': 86400,
315        'W': 604800,
316        'M': 2592000,
317        'Y': 31536000,
318    }
319
320    string = string.strip()
321    if not string or len(string) < 2:
322        return 0
323
324    unit = string[-1]
325    value_str = string[:-1]
326
327    if unit not in unit_to_seconds:
328        return 0
329
330    try:
331        value = float(value_str)
332    except (ValueError, TypeError):
333        return 0
334
335    return int(value * unit_to_seconds[unit])

338def humanduration2seconds(text):
339    """
340    Converts a complex human-readable duration string into seconds by summing individual durations.
341
342    This function processes a string that may contain multiple duration components
343    (e.g., '3Y 2M 7s') and converts each component into seconds. It ignores non-valid components
344    (e.g., 'any-error') and sums the valid ones.
345
346    ### Parameters
347    - **text** (`str`): A string containing one or more human-readable durations, where each
348      duration is represented by a number followed by a unit (e.g., '3Y', '2M', '7s').
349      Invalid components are ignored.
350
351    ### Returns
352    - **int**: The total duration in seconds, summing all valid duration components. Returns 0 if
353      no valid components are found.
354
355    ### Example
356    >>> text = '3Y 2M any-error 3d7s'  # means: valid is '3Y 2M 7s'
357    >>> humanduration2seconds(text)
358    99792007
359    """
360    return sum(human2seconds(duration) for duration in extract_hrnumbers(text))

363def humanrange2bytes(text):
364    """
365    Converts a Nagios range (e.g., `@4K:5 MiB`) into a range in bytes, where the base is always
366    1024.
367
368    This function processes a Nagios-style range string that may contain units such as 'K', 'M',
369    'B', etc., and converts each part into bytes using the `human2bytes` function. The result is
370    a string with the range values expressed in bytes.
371
372    ### Parameters
373    - **text** (`str`): A Nagios-style range string, such as '@4K:5 MiB', where units like K, M,
374      or B are used.
375
376    ### Returns
377    - **str**: The range with each value converted into bytes, using 1024 as the base for 
378      conversions.
379
380    ### Example
381    >>> text = '@4K:5 MiB'
382    >>> humanrange2bytes(text)
383    '@4096:5242880'
384    """
385    parts = []
386    for part in text.split(':'):
387        if not part:
388            continue
389        size = part.replace('-', '').replace('~', '').replace('@', '')
390        size_bytes = human2bytes(size)
391        parts.append(part.replace(size, str(size_bytes)))
392
393    result = ':'.join(parts)
394    if text.startswith(':'):
395        result = ':' + result
396    if text.endswith(':'):
397        result += ':'
398    return result

401def humanrange2seconds(string):
402    """
403    Converts a Nagios range to seconds by interpreting the duration components and summing them.
404
405    This function processes a Nagios-style range string (e.g., `@10m:1Y1D`) and converts each part
406    into seconds using the `humanduration2seconds` function. The result is a string with the range
407    values expressed in seconds.
408
409    ### Parameters
410    - **string** (`str`): A Nagios-style range string, such as '@10m:1Y1D', where units like 'm',
411      'Y', 'D' are used.
412
413    ### Returns
414    - **str**: The range with each value converted into seconds.
415
416    ### Example
417    >>> string = '@10m:1Y1D'
418    >>> humanrange2seconds(string)
419    '@600:31622400'
420    """
421    result = []
422    for part in string.split(':'):
423        if not part:
424            continue
425        duration = part.replace('-', '').replace('~', '').replace('@', '')
426        seconds = humanduration2seconds(duration)
427        result.append(part.replace(duration, f'{seconds}'))
428
429    if string.startswith(':'):
430        return ':' + ':'.join(result)
431    if string.endswith(':'):
432        return ':'.join(result) + ':'
433    return ':'.join(result)

436def number2human(number):
437    """
438    Converts a number into a human-readable format using SI prefixes.
439
440    This function converts large numbers into a more concise format with SI prefixes
441    (e.g., 1,000 becomes '1K', 1,000,000 becomes '1M'). It supports values from 1 to extremely large
442    numbers (up to 'Y' for 10^24).
443
444    ### Parameters
445    - **number** (`int` or `float`): The number to convert into a human-readable format.
446
447    ### Returns
448    - **str**: The number formatted with an appropriate SI prefix.
449
450    ### Example
451    >>> number2human(123456.8)
452    '123.5K'
453
454    >>> number2human(123456789.0)
455    '123.5M'
456
457    >>> number2human(9223372036854775808)
458    '9.2E'
459    """
460    millnames = ['', 'K', 'M', 'G', 'T', 'P', 'E', 'Z', 'Y']
461    try:
462        number = float(number)
463    except Exception:
464        return number
465
466    if number == 0:
467        millidx = 0
468    else:
469        millidx = int(math.floor(math.log10(abs(number)) / 3))
470
471    millidx = max(0, min(len(millnames) - 1, millidx))
472    scaled = number / 10**(3 * millidx)
473    return f'{scaled:.1f}{millnames[millidx]}'

476def seconds2human(seconds, keep_short=True, full_name=False):
477    """
478    Converts a number of seconds into a human-readable time range string.
479
480    This function takes a number of seconds and returns a string that expresses that duration in
481    a more understandable format. It supports both short and full-form time units (e.g., "1m" for
482    minutes, "1 hour 30 minutes" for the full form).
483
484    ### Parameters
485    - **seconds** (`int` or `float` or `str`): The number of seconds to convert.
486    - **keep_short** (`bool`, optional): If True, returns only the largest two time units
487      (default is True).
488    - **full_name** (`bool`, optional): If True, returns full names for the time units
489      (default is False).
490
491    ### Returns
492    - **str**: The formatted time duration in a human-readable format.
493
494    ### Example
495    >>> seconds2human(0.125)
496    '0.12s'
497
498    >>> seconds2human(1)
499    '1s'
500
501    >>> seconds2human(59)
502    '59s'
503
504    >>> seconds2human(60)
505    '1m'
506
507    >>> seconds2human(61)
508    '1m 1s'
509
510    >>> seconds2human(1387775)
511    '2W 2D'
512
513    >>> seconds2human('1387775')
514    '2W 2D'
515
516    >>> seconds2human('1387775', full_name=True)
517    '2weeks 2days'
518
519    >>> seconds2human(1387775, keep_short=False, full_name=True)
520    '2weeks 2days 1hour 29minutes 35seconds'
521    """
522    seconds = float(seconds)
523
524    units = (
525        ('years', 60 * 60 * 24 * 365),
526        ('months', 60 * 60 * 24 * 30),
527        ('weeks', 60 * 60 * 24 * 7),
528        ('days', 60 * 60 * 24),
529        ('hours', 60 * 60),
530        ('minutes', 60),
531        ('seconds', 1),
532        ('millisecs', 1e-3),
533        ('microsecs', 1e-6),
534        ('nanosecs', 1e-9),
535        ('picosecs', 1e-12),
536    ) if full_name else (
537        ('Y', 60 * 60 * 24 * 365),
538        ('M', 60 * 60 * 24 * 30),
539        ('W', 60 * 60 * 24 * 7),
540        ('D', 60 * 60 * 24),
541        ('h', 60 * 60),
542        ('m', 60),
543        ('s', 1),
544        ('ms', 1e-3),
545        ('us', 1e-6),
546        ('ns', 1e-9),
547        ('ps', 1e-12),
548    )
549
550    result = []
551    for name, count in units:
552        value = seconds // count
553        if value:
554            seconds -= value * count
555            if full_name and value == 1:
556                name = name.rstrip('s')
557            result.append(f'{int(value)}{name}')
558
559    if keep_short and len(result) > 2:
560        return ' '.join(result[:2])
561    return ' '.join(result)
