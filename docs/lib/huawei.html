100def get_creds(args):
101    """
102    Retrieve and cache Huawei appliance credentials.
103
104    This function handles authentication against a Huawei device API. It reuses cached tokens
105    (`iBaseToken` and `cookie`) if available to avoid repeated logins, which may be rate-limited for
106    security reasons. If no cached credentials are found, it performs a login request and caches
107    the new credentials for future reuse.
108
109    ### Parameters
110    - **args** (object):
111      An argument object containing:
112        - `URL` (`str`): Base URL of the Huawei API.
113        - `DEVICE_ID` (`str`): Unique device identifier.
114        - `USERNAME` (`str`): Login username.
115        - `PASSWORD` (`str`): Login password.
116        - `SCOPE` (`str`): Authentication scope.
117        - `INSECURE` (`bool`): Whether to disable SSL verification.
118        - `NO_PROXY` (`bool`): Whether to ignore proxy settings.
119        - `TIMEOUT` (`int`): Request timeout in seconds.
120        - `CACHE_EXPIRE` (`int`): Cache expiration time in minutes.
121
122    ### Returns
123    - **tuple** (`str`, `str`):
124      - `ibase_token` (str): The API session token (iBaseToken).
125      - `cookie` (str): The session cookie.
126
127    ### Notes
128    - Tokens are stored in cache keys:
129      - `huawei-{DEVICE_ID}-ibasetoken`
130      - `huawei-{DEVICE_ID}-cookie`
131    - If login is required, the request is sent as serialized JSON with headers.
132
133    ### Example
134    >>> ibasetoken, cookie = get_creds(args)
135    """
136    token_key = f'huawei-{args.DEVICE_ID}-ibasetoken'
137    cookie_key = f'huawei-{args.DEVICE_ID}-cookie'
138
139    ibasetoken = cache.get(token_key)
140    cookie = cache.get(cookie_key)
141
142    if ibasetoken:
143        return ibasetoken, cookie
144
145    uri = f'{args.URL}/deviceManager/rest/{args.DEVICE_ID}/sessions'
146    header = {'Content-Type': 'application/json'}
147    data = {
148        'username': args.USERNAME,
149        'password': args.PASSWORD,
150        'scope': args.SCOPE,
151    }
152    result = base.coe(url.fetch_json(
153        uri,
154        data=data,
155        encoding='serialized-json',
156        extended=True,
157        header=header,
158        insecure=args.INSECURE,
159        no_proxy=args.NO_PROXY,
160        timeout=args.TIMEOUT,
161    ))
162
163    ibasetoken = result.get('response_json', {}).get('data', {}).get('iBaseToken')
164    cookie = result.get('response_header', {}).get('Set-Cookie')
165
166    expire = time.now() + args.CACHE_EXPIRE * 60
167    cache.set(token_key, ibasetoken, expire)
168    cache.set(cookie_key, cookie, expire)
169
170    return ibasetoken, cookie

173def get_data(endpoint, args, params=''):
174    """
175    Fetch data from a Huawei appliance endpoint, with automatic retry on authorization errors.
176
177    This function performs an authenticated GET request to a Huawei device's REST API. It handles
178    common transient errors (like unauthorized errors) by retrying the request several times before
179    giving up.
180
181    ### Parameters
182    - **endpoint** (`str`):
183      The API endpoint to call (relative path after the device ID).
184    - **args** (object):
185      An object containing:
186        - `URL` (`str`): Base API URL.
187        - `DEVICE_ID` (`str`): Device ID.
188        - `INSECURE` (`bool`): Disable SSL verification.
189        - `NO_PROXY` (`bool`): Ignore proxy settings.
190        - `TIMEOUT` (`int`): Timeout for API requests.
191    - **params** (`str`, optional):
192      Additional URL parameters (starting with `?`, if any). Default is empty.
193
194    ### Returns
195    - **dict**:
196      The parsed JSON response from the API, plus an extra `counter` key showing retry attempts.
197
198    ### Notes
199    - Automatically retries up to 9 times on authorization failures (`-401` errors).
200    - Waits 1 second between retries.
201
202    ### Example
203    >>> get_data('disk/list', args)
204    {
205        'error': {'code': 0},
206        'data': {...},
207        'counter': 1
208    }
209    """
210    ibasetoken, cookie = get_creds(args)
211
212    uri = f'{args.URL}/deviceManager/rest/{args.DEVICE_ID}/{endpoint}{params}'
213    header = {
214        'Content-Type': 'application/json',
215        'iBaseToken': ibasetoken,
216        'Cookie': cookie,
217    }
218
219    max_retries = 9
220    counter = 0
221
222    for attempt in range(1, max_retries + 1):
223        result = base.coe(url.fetch_json(
224            uri,
225            header=header,
226            insecure=args.INSECURE,
227            no_proxy=args.NO_PROXY,
228            timeout=args.TIMEOUT,
229        ))
230        counter = attempt
231        if result.get('error', {}).get('code') == 0:
232            break
233        _time.sleep(1)
234
235    result['counter'] = counter
236    return result
