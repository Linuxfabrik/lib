 80def cpu_cores(dmi):
 81    """
 82    Calculate the total number of CPU cores.
 83
 84    This function sums the core count from all processor entries in the given DMI data structure.
 85    If a processor entry does not specify a core count, it is treated as zero.
 86
 87    ### Parameters
 88    - **dmi** (`dict`):
 89      The parsed DMI data, typically a dictionary from SMBIOS or `dmidecode` output.
 90
 91    ### Returns
 92    - **int**:
 93      The total number of CPU cores found across all processors.
 94
 95    ### Notes
 96    - Entries are retrieved using `dmiget(dmi, 'Processor')`.
 97    - Missing or invalid core counts default to zero.
 98
 99    ### Example
100    >>> cpu_cores(parsed_dmi)
101    8
102    """
103    return sum(int(cpu.get('Core Count', 0)) for cpu in dmiget(dmi, 'Processor'))

132def cpu_speed(dmi):
133    """
134    Retrieve the CPU speed in megahertz (MHz).
135
136    This function checks all processor entries in the given DMI data and returns the speed
137    of the first valid CPU found. Speeds are expected to be specified in MHz.
138
139    ### Parameters
140    - **dmi** (`dict`):
141      The parsed DMI data, typically a dictionary from SMBIOS or `dmidecode` output.
142
143    ### Returns
144    - **int**:
145      The CPU speed in MHz. If no valid entry is found, returns `0`.
146
147    ### Notes
148    - Entries without a proper `Current Speed` field are skipped.
149    - Only the first valid speed encountered is returned.
150
151    ### Example
152    >>> cpu_speed(parsed_dmi)
153    3200
154    """
155    for cpu in dmiget(dmi, 'Processor'):
156        current_speed = cpu.get('Current Speed', '')
157        if current_speed.endswith(' MHz'):
158            return int(current_speed.replace(' MHz', '').strip())
159    return 0

188def cpu_type(dmi):
189    """
190    Retrieve the CPU type string.
191
192    This function extracts the CPU type from the `Version` field of the first enabled processor
193    entry in the given DMI data. Trademark symbols like `(R)` and `(TM)` are removed for clarity.
194
195    ### Parameters
196    - **dmi** (`dict`):
197      The parsed DMI data, typically a dictionary from SMBIOS or `dmidecode` output.
198
199    ### Returns
200    - **str**:
201      The CPU type as a clean string. Returns `'n/a'` if no valid CPU entry is found.
202
203    ### Notes
204    - The `Version` field is sanitized to remove `(R)`, `(TM)`, and `'NotSpecified'` strings.
205    - Only the first enabled CPU entry is considered.
206
207    ### Example
208    >>> cpu_type(parsed_dmi)
209    'Intel Xeon Silver 4210 CPU'
210    """
211    for cpu in dmiget(dmi, 'Processor'):
212        if int(cpu.get('Core Enabled', 0)) > 0:
213            cpu_version = cpu.get('Version', '').replace('(R)', '').replace('(TM)', '').strip()
214            if 'NotSpecified' in cpu_version or not cpu_version:
215                return 'n/a'
216            return cpu_version
217    return 'n/a'

220def dmidecode_parse(output):
221    """
222    Parse the raw output of `dmidecode` into a structured dictionary.
223
224    This function processes the raw textual output of the `dmidecode` tool and extracts
225    structured information about system hardware, organized by DMI handle.
226
227    ### Parameters
228    - **output** (`str`):
229      The raw string output from the `dmidecode` command.
230
231    ### Returns
232    - **dict**:
233      A dictionary keyed by DMI handle. Each value contains fields such as dminame, dmisize,
234      dmitype, and parsed key-value pairs from the output.
235
236    ### Notes
237    - Records are separated by double newlines in the output.
238    - Only records with at least three lines are considered valid.
239    - Multi-line blocks are handled if needed.
240
241    ### Example
242    >>> dmidecode_parse(dmidecode_output)
243    {
244        ('0xDA00', '218', '251'): {
245            'dminame': 'OEM-specific Type',
246            'dmisize': 251,
247            'dmitype': 218,
248            'H': 'D\t\t0\t\t0\t\t0\t\t0\t\t0\t\t0\t\t0\t\t0\t\t0\t\t0\t\t0\t\t0\t\t0\t\t0\t\t0'
249        },
250        ('0x0001', '0', '26'): {
251            'dminame': 'BIOS Information',
252            'dmisize': 26,
253            'dmitype': 0,
254            'Vendor': 'Dell Inc.',
255            'Version': '1.7.1',
256            'Release Date': '12/06/2024',
257            'ROM Size': '64 MB',
258            ...,
259        },
260        ...
261    }
262    """
263    data = {}
264
265    for record in output.split('\n\n'):
266        record_element = record.splitlines()
267        if len(record_element) < 3:
268            continue
269
270        handle_data = HANDLE_RE.findall(record_element[0])
271        if not handle_data:
272            continue
273
274        dmi_handle = handle_data[0]
275        data[dmi_handle] = {
276            'dminame': record_element[1],
277            'dmisize': int(dmi_handle[2]),
278            'dmitype': int(dmi_handle[1]),
279        }
280
281        in_block_element = None
282        in_block_list = []
283
284        for line in record_element[2:]:
285            if in_block_element:
286                in_block_data = IN_BLOCK_RE.findall(line)
287                if in_block_data:
288                    in_block_list.append(in_block_data[0][0])
289                    data[dmi_handle][in_block_element] = '\t\t'.join(in_block_list)
290                    continue
291                else:
292                    in_block_element = None
293                    in_block_list = []
294
295            record_data = RECORD_RE.findall(line)
296            if record_data:
297                key, value = record_data[0]
298                data[dmi_handle][key] = value
299                continue
300
301            record_data2 = RECORD2_RE.findall(line)
302            if record_data2:
303                in_block_element = record_data2[0][0]
304                in_block_list = []
305
306    return data

309def dmiget(dmi, type_id):
310    """
311    Retrieve DMI entries of a specific type.
312
313    This function filters a parsed DMI data structure, returning all entries matching the specified
314    type ID. If a string type name is given, it is internally mapped to its numeric type ID.
315
316    ### Parameters
317    - **dmi** (`dict`):
318      The parsed DMI data, typically a dictionary from SMBIOS or `dmidecode` output.
319    - **type_id** (`int` or `str`):
320      The DMI type as an integer or string. If a string is given, it is matched against known
321      type names.
322
323    ### Returns
324    - **list**:
325      A list of DMI entries (dicts) matching the specified type ID.
326
327    ### Notes
328    - The `TYPE2STR` mapping must be available globally to resolve string type names.
329    - Useful for extracting subsets like BIOS information, baseboard details, or memory devices.
330
331    ### Example
332    >>> dmiget(parsed_dmi, 'Memory Device')
333    [{'Handle': '0x1100', 'Size': '8 GB', 'Form Factor': 'SODIMM', ...}]
334    """
335    if isinstance(type_id, str):
336        type_id = next(
337            (type_num for type_num, type_str in TYPE2STR.items() if type_str == type_id),
338            None
339        )
340        if type_id is None:
341            return []
342
343    return [item for item in dmi.values() if item.get('dmitype') == type_id]

374def get_data():
375    """
376    Retrieve and parse DMI data using `dmidecode`.
377
378    This function executes the `dmidecode` command, parses its output, and returns structured
379    DMI data. If execution fails or returns a non-zero exit code, returns `False`.
380
381    ### Parameters
382    - *(none)*
383
384    ### Returns
385    - **dict**:
386      Parsed DMI data if successful.
387
388    - **bool**:
389      `False` on failure (e.g., command failed, permission denied, `dmidecode` not installed).
390
391    ### Notes
392    - Requires root privileges to run `dmidecode`.
393    - Depends on a shell execution helper (`shell.shell_exec`).
394
395    ### Example
396    >>> get_data()
397    {
398        ('0xDA00', '218', '251'): {
399            'dminame': 'OEM-specific Type',
400            'dmisize': 251,
401            'dmitype': 218,
402            'H': 'D\t\t0\t\t0\t\t0\t\t0\t\t0\t\t0\t\t0\t\t0\t\t0\t\t0\t\t0\t\t0\t\t0\t\t0\t\t0'
403        },
404        ('0x0001', '0', '26'): {
405            'dminame': 'BIOS Information',
406            'dmisize': 26,
407            'dmitype': 0,
408            'Vendor': 'Dell Inc.',
409            'Version': '1.7.1',
410            'Release Date': '12/06/2024',
411            'ROM Size': '64 MB',
412            ...,
413        },
414        ...
415    }
416    """
417    success, result = shell.shell_exec('sudo dmidecode')
418    if not success:
419        return False
420
421    stdout, stderr, retc = result
422    if retc != 0:
423        return False
424
425    return dmidecode_parse(stdout)

484def ram(dmi):
485    """
486    Calculate the total amount of RAM installed in bytes.
487
488    This function sums the memory size of all populated memory slots found in the given
489    DMI data. Slot sizes given in megabytes (MB) or gigabytes (GB) are normalized to bytes.
490
491    ### Parameters
492    - **dmi** (`dict`):
493      The parsed DMI data, typically a dictionary from SMBIOS or `dmidecode` output.
494
495    ### Returns
496    - **int**:
497      The total amount of RAM in bytes.
498
499    ### Notes
500    - Only slots reporting a size in MB or GB are considered.
501    - Entries reporting "No module installed" are skipped.
502
503    ### Example
504    >>> ram(parsed_dmi)
505    34359738368
506    """
507    total = 0
508    for slot in dmiget(dmi, 'Memory Device'):
509        size_field = slot.get('Size', '').upper()
510        if 'GB' in size_field:
511            size = int(size_field.replace(' GB', '').strip()) * 1024**3
512            total += size
513        elif 'MB' in size_field:
514            size = int(size_field.replace(' MB', '').strip()) * 1024**2
515            total += size
516    return total
