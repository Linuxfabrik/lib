 30def check_select_privileges(conn):
 31    """
 32    Verify if the database connection has sufficient privileges to run SELECT statements.
 33
 34    This function attempts a simple `SELECT version()` query against the connected database.
 35    If the query succeeds and returns results, SELECT privileges are assumed to be granted.
 36    Otherwise, it indicates insufficient permissions.
 37
 38    ### Parameters
 39    - **conn** (`mysql.connector.connection.MySQLConnection` or similar):
 40      An active database connection object.
 41
 42    ### Returns
 43    - **tuple** (`bool`, `any`):
 44      - First element (`bool`): `True` if SELECT succeeded, `False` otherwise.
 45      - Second element (`any`): Query result if successful, or an error message string if failed.
 46
 47    ### Notes
 48    - A failure usually indicates missing `SELECT` privileges on the connected database user.
 49    - The actual result returned on success is the database server's version string inside a
 50      dictionary.
 51
 52    ### Example
 53    >>> success, version = check_select_privileges(conn)
 54    >>> if success:
 55    >>>     print(f"Connected to MySQL version: {version['version']}")
 56    >>> else:
 57    >>>     print(version)
 58    """
 59    success, result = select(conn, 'SELECT VERSION() AS version;', fetchone=True)
 60    if success and result:
 61        return True, result
 62    return False, 'You probably do not have sufficient privileges to run SELECT statements.'

 96def commit(conn):
 97    """
 98    Commit any pending changes to the database.
 99
100    This function saves (commits) all changes made during the current database session.
101    If the commit fails, it returns an error message.
102
103    ### Parameters
104    - **conn** (`Connection`):
105      An active database connection object.
106
107    ### Returns

132def connect(mysql_connection, **kwargs):
133    """
134    Connect to a MySQL or MariaDB server using a dictionary of connection parameters.
135
136    This function establishes a database connection using parameters from the given dictionary,
137    such as configuration file, group, timeout, and cursor class. Additional connection options
138    can be passed via `**kwargs`.
139
140    ### Parameters
141    - **mysql_connection** (`dict`):
142      A dictionary containing connection settings.
143      - `defaults_file` (`str`, optional): Path to a MySQL options file.
144      - `defaults_group` (`str`, optional): Group to read from the options file.
145        Defaults to `'client'`.
146      - `timeout` (`int`, optional): Connection timeout in seconds. Defaults to `3`.
147      - `cursorclass` (optional): Cursor class to use. Defaults to `DictCursor`.
148    - **kwargs** (`any`, optional):
149      Additional keyword arguments passed directly to `pymysql.connect()`.
150
151    ### Returns
152    - **tuple** (`bool`, `Connection or str`):
153      - First element (`bool`): `True` if connection succeeded, `False` if it failed.
154      - Second element (`Connection` or `str`):
155        - Database connection object on success.
156        - Error message string on failure.
157
158    ### Notes
159    - If connection fails, the error message contains the reason for failure.
160    - `pymysql`'s `read_default_file` and `read_default_group` allow connection settings from a
161      `.cnf` file.
162
163    ### Example
164    >>> mysql_connection = {
165    >>>     'defaults_file': '/etc/mysql/my.cnf',
166    >>>     'defaults_group': 'client',
167    >>>     'timeout': 5,
168    >>> }
169    >>> success, conn = connect(mysql_connection)
170    >>> if success:
171    >>>     # Use conn
172    >>>     pass
173    >>> else:
174    >>>     print(conn)
175    """
176    try:
177        conn = pymysql.connect(
178            read_default_file=mysql_connection.get('defaults_file'),
179            read_default_group=mysql_connection.get('defaults_group', 'client'),
180            cursorclass=mysql_connection.get('cursorclass', pymysql.cursors.DictCursor),
181            connect_timeout=mysql_connection.get('timeout', 3),
182            **kwargs,
183        )
184        return True, conn
185    except Exception as e:
186        return False, f'Connecting to DB failed: {e}'

189def get_engines(conn):
190    """
191    Retrieve the available storage engines from the database.
192
193    This function runs `SHOW ENGINES` against the connected database, normalizes the output,
194    and returns a dictionary mapping engine names to their support status.
195    It emulates the old `have_*` status variables for compatibility with older codebases.
196    Also works around MySQL bug #59393 related to `skip-innodb`.
197
198    ### Parameters
199    - **conn** (`Connection`):
200      An active database connection object.
201
202    ### Returns
203    - **dict** (`str: str`):
204      A dictionary where keys are `have_<engine>` and values are:
205      - `'YES'` if the engine is available by default.
206      - The actual support status reported otherwise (`DISABLED`, etc.).
207
208    ### Notes
209    - `have_*` variables are deprecated since MySQL 5.6 and removed afterward.
210    - Special mappings:
211      - `federated`  `have_federated_engine`
212      - `blackhole`  `have_blackhole_engine`
213      - `berkeleydb`  `have_bdb`
214    - This function helps maintain compatibility with monitoring scripts expecting old-style
215      `have_*` variables.
216
217    ### Example
218    >>> get_engines(conn)
219    {
220        'have_innodb': 'YES',
221        'have_myisam': 'YES',
222        'have_blackhole_engine': 'DISABLED',
223        ...
224    }
225    """
226    engines = {}
227    success, result = select(conn, 'SHOW ENGINES')
228
229    if not success or not result:
230        return engines
231
232    for line in result:
233        engine = line['Engine'].lower()
234        if engine in ('federated', 'blackhole'):
235            engine += '_engine'
236        elif engine == 'berkeleydb':
237            engine = 'bdb'
238
239        engines[f'have_{engine}'] = 'YES' if line['Support'] == 'DEFAULT' else line['Support']
240
241    return engines

244def lod2dict(lod):
245    """
246    Convert a list of simple key-value dictionaries into a single dictionary.
247
248    This function processes a list of dictionaries and merges them into one dictionary.
249    It handles special cases where the input uses `Variable_name` and `Value` fields, as returned
250    by SQL queries like `SHOW VARIABLES;`.
251
252    ### Parameters
253    - **lod** (`list` of `dict`):
254      A list where each element is a dictionary with either:
255      - A simple `{key: value}` structure.
256      - A special `{Variable_name: ..., Value: ...}` structure (from MySQL system queries).
257
258    ### Returns
259    - **dict** (`str: str`):
260      A dictionary with keys and values extracted from the input list.
261
262    ### Notes
263    - If a dictionary contains `Variable_name` and `Value`, they are used as key and value.
264    - Otherwise, the first key-value pair from the dictionary is used directly.
265    - Later keys will overwrite earlier ones if duplicate keys exist.
266
267    ### Example
268    >>> lod2dict([{'Variable_name': 'a', 'Value': 'b'}, {'Variable_name': 'c', 'Value': 'd'}])
269    {'a': 'b', 'c': 'd'}
270
271    >>> lod2dict([{'key1': 'value1'}, {'key2': 'value2'}])
272    {'key1': 'value1', 'key2': 'value2'}
273    """
274    result = {}
275    for row in lod:
276        if 'Variable_name' in row and 'Value' in row:
277            result[row['Variable_name']] = row['Value']
278        else:
279            result.update(row)
280    return result

283def select(conn, sql, data=None, fetchone=False):
284    """
285    Execute a SQL SELECT query on the database connection.
286
287    This function executes a SELECT statement against the connected database,
288    optionally using provided parameters, and returns either one row or all rows.
289    SELECT operations do not modify the database.
290
291    ### Parameters
292    - **conn** (`Connection`):
293      An active database connection object.
294    - **sql** (`str`):
295      The SQL SELECT query to execute.
296      Use placeholders (`%s`) for any parameters.
297    - **data** (`list`, optional):
298      A list of values to bind to the placeholders in the SQL query.
299      Defaults to an empty list (no parameters).
300    - **fetchone** (`bool`, optional):
301      If `True`, fetch only the first matching row.
302      If `False` (default), fetch all matching rows.
303
304    ### Returns
305    - **tuple** (`bool`, `any`):
306      - First element (`bool`): `True` if the query succeeded, `False` if it failed.
307      - Second element (`list`, `dict`, or `str`):
308        - The query result (one row as a dict if `fetchone=True`, or a list of dicts).
309        - Error message string on failure.
310
311    ### Notes
312    - On success, results are returned as dictionaries (one per row) if the connection uses
313      `DictCursor`.
314    - On failure, an error message is returned with the failed SQL, the exception, and any
315      input data.
316
317    ### Example
318    Query using a LIKE pattern:
319    >>> data = ['val1%']
320    >>> sql = 'SELECT * FROM t WHERE c LIKE %s'
321    >>> success, result = select(conn, sql, data)
322
323    Query using an IN clause:
324    >>> data = ['val1', 'val2']
325    >>> sql = 'SELECT * FROM t WHERE c IN (f{", ".join("%s" for _ in data)})'
326    >>> success, result = select(conn, sql, data)
327    """
328    data = data or []
329    try:
330        with conn.cursor() as cursor:
331            cursor.execute(sql, tuple(data)) if data else cursor.execute(sql)
332            return (True, cursor.fetchone()) if fetchone else (True, cursor.fetchall())
333    except Exception as e:
334        return False, f'Query failed: {sql}, Error: {e}, Data: {data}'

This function executes a SELECT statement against the connected database, optionally using provided parameters, and returns either one row or all rows. SELECT operations do not modify the database.

Parameters

  • conn (Connection): An active database connection object.
  • sql (str): The SQL SELECT query to execute. Use placeholders (%s) for any parameters.
  • data (list, optional): A list of values to bind to the placeholders in the SQL query. Defaults to an empty list (no parameters).

Returns

  • tuple (bool, any):
